on_new_ruler = {
	effect = {
		if = {
			limit  ={
				owner = {has_law = law_type:law_autocracy}
			}
			owner = {
				every_interest_group = {
					remove_modifier = autocracy_power
				}
			}
			interest_group ={
				if = {
					limit = {
						not = {
							has_modifier = autocracy_power
						}
					}
					add_modifier = {
						name = autocracy_power
						#days = -1
					}
	
				}
			}
			
			
		}
	}

}
